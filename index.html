processTETranscriptsTable <- function(file) {
    
    # This function receives as input a sigdiff_gene_TE.txt file and returns a new dataframe with correctly processed ensemblIDs.
    # Next, it subsets those genes with a log2foldchange greater than or equal to 0.56 for downstream analysis.
    # Then, it retrieves geneNames using the API from biotools.fr. 
    # Finally, it saves the processed dataframe as a csv with a meaningful name.
    # ---------------------------------------
    
    # Read .txt file
    df <- read.table(
      file,
      header = TRUE,
      row.names = NULL,
      sep = "\t"   # use "," if comma-separated
    )
    
    # Rename first column to "ensemblID"
    colnames(df)[1] <- "ensemblID"
    
    # Create regular foldChange 
    df <- df %>%
      mutate(
        foldChange   = 2^log2FoldChange
      )
    
    # Remove dot "." and following characters from ensemblID column
    df <- df %>%
      mutate(ensemblID = str_remove(ensemblID, "\\..*"))
    
    # Create df_subset, a new df that only includes sigdiff_gene_TE's >= 0.56
    df_subset <- df %>%
      filter(log2FoldChange >= 0.56)
    
    # Create df_sorted, which sorts df_subset descending by log2FoldChange
    df_sorted <- df_subset %>%
      arrange(desc(log2FoldChange))
    
    # Get Gene Names
    # We used this Tool to convert these ensembl IDs to gene names: https://www.biotools.fr/mouse/ensembl_symbol_converter 
    # Optional: uncomment print and message statements to show progress
    print("Getting gene names via biotools.fr API")
    
    # Initialize progress bar
    pb <- txtProgressBar(min = 0, max = nrow(df_sorted), style = 3)
    
    for (i in seq_len(nrow(df_sorted))) {
      # message("Processing row ", i)
         
      # Brute force -- Use API in a loop
      id = df_sorted$ensemblID[i]
      url = paste("http://biotools.fr/mouse/ensembl_symbol_converter/?api=1&id=",id)
      r <- GET(url)
    
      output = fromJSON( content(r, "text", encoding = "UTF-8"), flatten=TRUE)
      # print(output)
    
      df <- as.data.frame(output)
        # print(df)
    
      if (nrow(df) == 0) {
          # message("No Gene Name for ", df_sorted$ensemblID[i])
          df_sorted$geneName[i] <- "N/A"
          }
      else {
          # message("Gene Name for ", df_sorted$ensemblID[i], ": ", df[1,1])    
          df_sorted$geneName[i] <- df[1,1]
          }  
        
      setTxtProgressBar(pb, i)
    }
    close(pb)
    
    print("Successfully got gene names")
    
    # Move geneNames col to first position
    df_final <- df_sorted %>%
      relocate(geneName)
    
    # Show df_final
    print("Head of result dataframe:")
    print(head(df_final))
    
    # Write csv in the current working directory
    fName = sub("\\.txt$", "", file)
    csvName = paste0(fName, "_Log2Fold0_56Sorted.csv")
    write.csv(df_final, file = csvName, row.names = FALSE)
    print(paste0("Successfully wrote ",csvName))
    
    # Return df_final
    df_final
}
